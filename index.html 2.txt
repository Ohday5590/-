        function handleKeyboard(e) {
            if (document.getElementById('lock-screen').style.display !== 'none') {
                if (e.key >= '0' && e.key <= '9') pressKey(e.key);
                if (e.key === 'Backspace') deleteLast();
                if (e.key === 'Escape') clearPass();
                return;
            }
            if (e.key === "Escape") { if (isModalOpen()) closeModals(); else if (document.getElementById('pos-screen').style.display === 'flex') goHome(); }
            if (e.ctrlKey && e.key === 's') { e.preventDefault(); saveDataToLocal(false); }
            if (e.key === "F1") { e.preventDefault(); if (document.getElementById('table-screen').style.display !== 'none') { activeTbl = 1; openPOS(); } else { document.getElementById('food-search').focus(); } }
            if(e.key==="Enter" && document.getElementById('pay-modal').style.display==='flex' && !document.getElementById('confirm-pay-btn').disabled) { finalizePay(); }
        }

        function exportHistoryToExcel() {
            if (saleHistory.length === 0) return alert("ບໍ່ມີຂໍ້ມູນການຂາຍ!");
            const excelData = saleHistory.flatMap(h => h.items.map(i => ({
                "ວັນທີ": h.date, "ເວລາ": h.time, "ໂຕ໊ະ": h.table, "ລາຍການ": i.name, "ລາຄາ": i.price, "ຈຳນວນ": i.qty, "ລວມ": i.price * i.qty
            })));
            const ws = XLSX.utils.json_to_sheet(excelData);
            const wb = XLSX.utils.book_new();
            XLSX.utils.book_append_sheet(wb, ws, "SalesHistory");
            XLSX.writeFile(wb, `POS_Report_${new Date().toLocaleDateString()}.xlsx`);
        }

        function openReport() {
            const today = new Date().toLocaleDateString();
            let todayTotal = 0; let grandTotal = 0; let menuStats = {};
            saleHistory.forEach(h => {
                grandTotal += h.total; if (h.date === today) todayTotal += h.total;
                h.items.forEach(item => { menuStats[item.name] = (menuStats[item.name] || 0) + item.qty; });
            });
            document.getElementById('rpt-today').innerText = todayTotal.toLocaleString() + " ₭";
            document.getElementById('rpt-total').innerText = grandTotal.toLocaleString() + " ₭";
            const topMenu = Object.entries(menuStats).sort((a, b) => b[1] - a[1]).slice(0, 5);
            const rptList = document.getElementById('rpt-top-menu');
            rptList.innerHTML = topMenu.length ? '' : '<p style="text-align:center; color:#888;">ຍັງບໍ່ມີຂໍ້ມູນການຂາຍ</p>';
            topMenu.forEach(([name, qty]) => { rptList.innerHTML += `<div class="top-menu-item"><span>${name}</span><b>x${qty}</b></div>`; });
            document.getElementById('report-modal').style.display = 'flex';
        }

        function handleFileSelect(event, type) {
            const file = event.target.files[0];
            if (file) {
                const reader = new FileReader();
                reader.onload = (e) => {
                    const res = e.target.result;
                    if (type === 'logo') config.logo = res;
                    else if (type === 'qr') config.qr = res;
                    else if (type === 'food') { document.getElementById('in-food-img').value = res; document.getElementById('food-img-preview').src = res; }
                    else if (type === 'bg') { document.documentElement.style.setProperty('--bg-img', `url(${res})`); }
                };
                reader.readAsDataURL(file);
            }
        }

        function saveDataToLocal(isAuto = false) {
            if(isAuto && !isAutoSaveEnabled) return;
            const data = { 
                tables, tableBooked, tableCount, menuData, config, isAutoSaveEnabled, saleHistory,
                correctPass,
                theme: document.body.hasAttribute('data-theme') ? 'dark' : 'light',
                bgSettings: { img: getComputedStyle(document.documentElement).getPropertyValue('--bg-img'), blur: document.getElementById('set-blur').value, bright: document.getElementById('set-bright').value }
            };
            localStorage.setItem('pos_ohday_data', JSON.stringify(data));
            if(isAuto) { const status = document.getElementById('autosave-status'); status.style.opacity = "1"; setTimeout(() => { status.style.opacity = "0"; }, 2000); }
        }

        function loadDataFromLocal() {
            const saved = localStorage.getItem('pos_ohday_data');
            if (saved) {
                const p = JSON.parse(saved);
                tables = p.tables || {}; tableBooked = p.tableBooked || {}; tableCount = p.tableCount || 12;
                menuData = p.menuData || menuData; config = p.config || config; saleHistory = p.saleHistory || [];
                correctPass = p.correctPass || "0000";
                isAutoSaveEnabled = p.isAutoSaveEnabled !== undefined ? p.isAutoSaveEnabled : true;
                document.getElementById('autosave-toggle').checked = isAutoSaveEnabled;
                if(p.theme === 'dark') { document.body.setAttribute('data-theme', 'dark'); document.getElementById('darkmode-toggle').checked = true; }
                if(p.bgSettings) {
                    document.documentElement.style.setProperty('--bg-img', p.bgSettings.img);
                    document.getElementById('set-blur').value = p.bgSettings.blur;
                    document.getElementById('set-bright').value = p.bgSettings.bright;
                    applyBgSettings();
                }
            }
            init();
        }

            document.documentElement.style.setProperty('--bg-blur', blur);
            document.documentElement.style.setProperty('--bg-brightness', bright);
        }

            document.documentElement.style.setProperty('--bg-img', 'none');

        function renderMenu() {
            const area = document.getElementById('menu-area');
            const search = document.getElementById('food-search').value.toLowerCase();
            area.innerHTML = '';
            menuData.filter(m => {
                const matchSearch = m.name.toLowerCase().includes(search);
                const matchCat = currentCategory === "ທັງໝົດ" || m.category === currentCategory;
                return matchSearch && matchCat;
            }).forEach(item => {
                const card = document.createElement('div');
                card.className = 'food-card';
                card.innerHTML = `<div class="food-actions"><button class="action-btn" style="background:#fbbc04" onclick="openEditFood(event, ${item.id})">✏️</button><button class="action-btn del-food-btn" onclick="deleteMenuItem(event, ${item.id})">×</button></div><img src="${item.img}" class="food-img" onerror="this.src='https://via.placeholder.com/150'"><div class="food-info"><b>${item.name}</b><br><span style="color:var(--primary); font-weight:900;">${item.price.toLocaleString()} ₭</span></div>`;
                card.onclick = () => {
                    const found = tables[activeTbl].find(f => f.id === item.id);
                    if(found) found.qty++; else tables[activeTbl].push({...item, qty:1});
                    if(tableBooked[activeTbl]) { tableBooked[activeTbl] = false; updateBookingBtn(); }
                    renderCart();
                };
                area.appendChild(card);
            });
        }

        function renderCart() {
            const list = document.getElementById('cart-list'); list.innerHTML = '';
            let total = 0;
            tables[activeTbl].forEach(item => {
                let lineTotal = item.price * item.qty; total += lineTotal;
                list.innerHTML += `<div class="cart-item"><div style="display:flex; justify-content:space-between;"><b>${item.name}</b><span>${lineTotal.toLocaleString()} ₭</span></div><div style="display:flex; align-items:center; gap:10px; margin-top:5px;"><button onclick="updateQty(${item.id}, -1)">-</button><span>${item.qty}</span><button onclick="updateQty(${item.id}, 1)">+</button><button onclick="removeItem(${item.id})" style="margin-left:auto; color:red; border:none; background:none; cursor:pointer;">🗑</button></div></div>`;
            });
            document.getElementById('total-display').innerText = total.toLocaleString() + " ₭";
        }

        function updateQty(id, delta) {
            const item = tables[activeTbl].find(i => i.id === id);
            if(item) { item.qty += delta; if(item.qty <= 0) removeItem(id); }
            renderCart();
        }
        function removeItem(id) { tables[activeTbl] = tables[activeTbl].filter(i => i.id !== id); renderCart(); }

        function openPay() {
            let total = tables[activeTbl].reduce((a,b)=>a+(b.price*b.qty), 0);
            if(total <= 0) return;
            document.getElementById('pay-modal').style.display='flex';
            document.getElementById('pay-total-val').innerText = total.toLocaleString();
            document.getElementById('pay-input').value = '';
            const qrImg = document.getElementById('qr-auto-img');
            const qrData = `bcelone://pay?merchantId=${config.merchantId}&amount=${total}&description=Table${activeTbl}`;
            qrImg.src = `https://api.qrserver.com/v1/create-qr-code/?size=200x200&data=${encodeURIComponent(qrData)}`;
            calcChange();
        }

        function calcChange() {
            let total = tables[activeTbl].reduce((a,b)=>a+(b.price*b.qty), 0);
            let inputVal = document.getElementById('pay-input').value;
            let cash = (parseFloat(inputVal) || 0) * 1000;
            let change = cash - total;
            const btn = document.getElementById('confirm-pay-btn');
            const display = document.getElementById('change-display');
            if(inputVal !== "" && change >= 0) { display.innerText = change.toLocaleString() + " ₭"; btn.disabled = false; btn.style.opacity = 1; }
            else { display.innerText = (inputVal === "" ? "0" : "ຂາດ " + Math.abs(change).toLocaleString()) + " ₭"; btn.disabled = true; btn.style.opacity = 0.5; }
        }

        function finalizePay() {
            const now = new Date();
            const orderNo = "ORD-" + Math.floor(1000 + Math.random() * 9000);
            let total = tables[activeTbl].reduce((a,b)=>a+(b.price*b.qty), 0);
            let cash = (parseFloat(document.getElementById('pay-input').value) || 0) * 1000;
            
            saleHistory.unshift({ 
                date: now.toLocaleDateString(), 
                time: now.getHours().toString().padStart(2, '0') + ":" + now.getMinutes().toString().padStart(2, '0'), 
                table: activeTbl, total: total, items: JSON.parse(JSON.stringify(tables[activeTbl]))

            const receiptArea = document.getElementById('receipt-print');
            let rowsHtml = tables[activeTbl].map((i, idx) => `

                    <td>${i.qty}</td>
                    <td>${i.price.toLocaleString()}</td>
                    <td style="text-align:right">${(i.price*i.qty).toLocaleString()}</td>

            receiptArea.innerHTML = `

                    <img src="${config.logo}" class="receipt-logo" onerror="this.style.display='none'">

                <table class="receipt-table">
                    <thead>
                        <tr>
                            <th>No</th>
                            <th>Item</th>
                            <th>Qty</th>
                            <th>Price</th>
                            <th style="text-align:right">Amount</th>
                        </tr>
                    </thead>
                    <tbody>${rowsHtml}</tbody>
                </table>
                <div class="receipt-line"></div>
                <div class="receipt-row"><b>ລວມທັງໝົດ / Total:</b><b>${total.toLocaleString()} ₭</b></div>

                document.getElementById('thank-you-popup').classList.remove('show'); 
                tables[activeTbl] = []; 
                tableBooked[activeTbl] = false;
