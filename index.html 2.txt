  const today = new Date().toDateString();
  if (state.lastDate !== today) {
    // shift history
    state.history = [...state.history.slice(1), 0];
    state.drunk = 0;
    state.lastDate = today;
    saveState();
  }

  updateDateTime();
  updateHydration();
  renderHistory();
  renderMotivation();
  renderWorkouts();
  renderVitamins();
  loadSettings();
  updateLang();
  simulateSteps();
}

function saveState() {
  localStorage.setItem('h2o_state', JSON.stringify(state));
}

// ===================== DATETIME =====================
function updateDateTime() {
  const t = i18n[state.lang];
  const now = new Date();
  const h = now.getHours();
  let greet = h < 12 ? t.greeting_morning : h < 18 ? t.greeting_afternoon : t.greeting_evening;
  if (state.name) greet = greet.replace('!', `, ${state.name}!`);
  document.getElementById('greeting-text').textContent = greet;
  const days = ['Sunday','Monday','Tuesday','Wednesday','Thursday','Friday','Saturday'];
  const months = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'];
  document.getElementById('date-text').textContent = `${days[now.getDay()]}, ${now.getDate()} ${months[now.getMonth()]} ${now.getFullYear()}`;
  
  // Weather hint based on "hour" (simulated)
  const wb = document.getElementById('weather-badge');
  const isHot = h >= 10 && h <= 17;
  wb.innerHTML = isHot 
    ? `<span>🌡️</span><span>${state.lang==='lo'?'35°C · ຮ້ອນ':state.lang==='en'?'35°C · Hot!':'35°C · ร้อน!'}</span>`
    : `<span>🌤️</span><span>${state.lang==='lo'?'28°C · ດີ':state.lang==='en'?'28°C · Nice':'28°C · ดี'}</span>`;
  wb.style.color = isHot ? 'var(--orange)' : 'var(--green)';
}

// ===================== HYDRATION =====================
function updateHydration() {
  const pct = Math.min(100, Math.round((state.drunk / state.goal) * 100));
  const remain = Math.max(0, state.goal - state.drunk);

  document.getElementById('cup-percent').textContent = pct + '%';
  document.getElementById('stat-drunk').textContent = state.drunk;
  document.getElementById('stat-remain').textContent = remain;
  document.getElementById('stat-goal').textContent = state.goal;

  // Water level: 0%=170 (bottom), 100%=10 (top) → y = 170 - (pct/100)*160
  const y = Math.round(170 - (pct / 100) * 160);
  document.getElementById('water-rect').setAttribute('y', y);
  
  // Update wave position
  const waveY = Math.max(y - 15, -10);
  const waveSvg = document.getElementById('wave-svg');
  if (waveSvg) waveSvg.setAttribute('y', waveY);
}

function addWater(ml) {
  state.drunk = Math.min(state.goal, state.drunk + ml);
  state.history[state.history.length - 1] = state.drunk;
  updateHydration();
  renderHistory();
  saveState();

  const t = i18n[state.lang];
  if (state.drunk >= state.goal) {
    showToast(t.toast_goal, 3000);
  } else {
    showToast(t.toast_added(ml) + ` ${state.drunk}ml`);
  }
  // ripple
  createRipple(event);
}

// ===================== HISTORY =====================
function renderHistory() {
  const container = document.getElementById('history-bars');
  const t = i18n[state.lang];
  const now = new Date();
  container.innerHTML = '';

  state.history.forEach((val, i) => {
    const dayOffset = i - 6;
    const d = new Date(now);
    d.setDate(d.getDate() + dayOffset);
    const dayName = t.days[d.getDay()];
    const pct = Math.min(100, Math.round((val / state.goal) * 100));
    const isToday = i === 6;
    const hue = pct >= 100 ? 'var(--green)' : pct >= 60 ? 'var(--cyan)' : pct >= 30 ? 'var(--blue-light)' : 'rgba(255,255,255,0.15)';

    container.innerHTML += `
      <div class="hbar-wrap">
        <div class="hbar" style="height:${Math.max(4, pct*0.5)}px;background:${isToday?'var(--cyan)':hue};${isToday?'box-shadow:0 0 10px var(--cyan-glow)':''}"></div>
        <div class="hbar-day" style="${isToday?'color:var(--cyan);font-weight:700':''}">
          ${dayName}
        </div>
      </div>`;
  });
}

// ===================== MOTIVATION =====================

  const motivs = i18n[state.lang].motivations;
  const pct = Math.min(100, Math.round((state.drunk / state.goal) * 100));
  let idx = Math.floor(pct / 25);
  if (idx >= motivs.length) idx = motivs.length - 1;
  document.getElementById('motivation-text').textContent = motivs[idx];
}

}

function closeWorkoutModalDirect() {
  clearInterval(timerInterval);
  timerRunning = false;
  document.getElementById('workout-modal').classList.remove('open');
}

function toggleTimer() {
  if (timerRunning) {
    clearInterval(timerInterval);
    timerRunning = false;
    document.getElementById('play-btn').textContent = '▶';
  } else {
    timerRunning = true;
    document.getElementById('play-btn').textContent = '⏸';
    timerInterval = setInterval(() => {
      timerSecs--;
      updateTimerDisplay();
      if (timerSecs <= 0) {
        clearInterval(timerInterval);
        timerRunning = false;
        document.getElementById('play-btn').textContent = '▶';
        showToast('🎉 ' + (state.lang === 'en' ? 'Workout done!' : state.lang === 'lo' ? 'ສຳເລັດ!' : 'ออกกำลังกายเสร็จแล้ว!'));
      }
    }, 1000);
  }
}

function resetTimer() {
  clearInterval(timerInterval);
  timerRunning = false;
  timerSecs = currentWorkout ? currentWorkout.duration : 300;
  document.getElementById('play-btn').textContent = '▶';
  updateTimerDisplay();
}

function updateTimerDisplay() {
  const m = Math.floor(timerSecs / 60);
  const s = timerSecs % 60;
  document.getElementById('timer-display').textContent = `${m}:${s.toString().padStart(2,'0')}`;
}

// ===================== SETTINGS =====================
function calcGoal() {
  const w = parseFloat(document.getElementById('input-weight').value) || 0;
  const act = parseFloat(document.getElementById('input-activity').value) || 1;
  if (w > 0) {
    // Formula: weight(kg) × 2.2 × 30 / 2 × activity_factor
    const ml = Math.round(w * 2.2 * 30 / 2 * act / 100) * 100;
    document.getElementById('calc-result-val').textContent = `${ml} ml`;
    state._calcGoal = ml;
  } else {
    document.getElementById('calc-result-val').textContent = '— ml';
    state._calcGoal = 0;
  }
}

function loadSettings() {
  if (state.name) document.getElementById('input-name').value = state.name;
  if (state.weight) document.getElementById('input-weight').value = state.weight;
  document.getElementById('profile-name-display').textContent = state.name || (state.lang === 'en' ? 'User' : state.lang === 'lo' ? 'ຜູ້ໃຊ້' : 'ผู้ใช้งาน');
  document.getElementById('profile-goal-display').textContent = (state.lang === 'en' ? 'Goal: ' : state.lang === 'lo' ? 'ເປົ້າ: ' : 'เป้าหมาย: ') + state.goal + ' ml/วัน';
  calcGoal();
}

function saveSettings() {
  state.name = document.getElementById('input-name').value;
  state.weight = parseFloat(document.getElementById('input-weight').value) || 0;
  if (state._calcGoal > 0) state.goal = state._calcGoal;
  saveState();
  loadSettings();
  updateDateTime();
  updateHydration();
  showToast(i18n[state.lang].toast_saved);
}

// ===================== STEPS (Simulated) =====================
function simulateSteps() {
  // Simulate step counting (in real app, use DeviceMotionEvent)
  setInterval(() => {
    if (Math.random() > 0.7) {
      state.steps = Math.min(6000, state.steps + Math.floor(Math.random() * 3) + 1);
      updateStepUI();
    }
  }, 2000);
}

function updateStepUI() {
  document.getElementById('step-count').textContent = state.steps.toLocaleString();
  const pct = Math.min(1, state.steps / 6000);
  const circumference = 239;
  document.getElementById('step-circle').setAttribute('stroke-dashoffset', circumference * (1 - pct));
  const kcal = Math.round(state.steps * 0.04);
  document.getElementById('kcal-text').textContent = i18n[state.lang].kcal(kcal);
}

function updateLang() {
  const t = i18n[state.lang];
  document.querySelectorAll('[data-i18n]').forEach(el => {
    const key = el.getAttribute('data-i18n');
    if (t[key] && typeof t[key] === 'string') el.textContent = t[key];
  });
  // Update step goal
  const sgEl = document.getElementById('step-goal-text');
  if (sgEl) sgEl.textContent = t.step_goal;
  // Search placeholder
  const si = document.getElementById('vit-search-input');
  if (si) si.placeholder = state.lang === 'en' ? 'Search vitamins...' : state.lang === 'lo' ? 'ຄົ້ນຫາ...' : 'ค้นหาวิตามิน...';
}

function showToast(msg, duration = 2000) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.classList.add('show');
  setTimeout(() => t.classList.remove('show'), duration);
}

function createRipple(e) {
  if (!e || !e.clientX) return;
  const r = document.createElement('div');
  r.className = 'ripple';
  r.style.cssText = `left:${e.clientX - 30}px;top:${e.clientY - 30}px;width:60px;height:60px`;
  document.body.appendChild(r);
  setTimeout(() => r.remove(), 600);
}
